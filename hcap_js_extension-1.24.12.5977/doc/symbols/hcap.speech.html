         console.log("onSuccess");
     },
     "onFailure" : function(f) {
         console.log("onFailure : errorMessage = " + f.errorMessage);
     }
});
    Parameters:
    {Object} param
    • {String} param.text [Required] - Indicates the text to be converted.
    • {Boolean} param.clear [Optional] - Indicates if the current text is to be cleared. true : Stops processing the current text, and processes the next one. false : Processes the next text only after completing the current one. (default : false)
    • {String} param.language [Optional] - Indicates the language used to read the input text. To get the list supported languages, use the hcap.speech.getSupportLanguagesTTS method. The default language is English (en-US).

    • {String} param.speed [Optional] - Indicates the speech speed of the input text. (default : medium) [verySlow | slow | medium | fast | veryFast]
